from diffusers import StableDiffusionPipeline
import torch
from google.colab import drive
from IPython.display import display
from datetime import datetime
import os

# Підключаємо Drive
drive.mount('/content/drive', force_remount=True)

# Завантажуємо модель
pipe = StableDiffusionPipeline.from_pretrained("runwayml/stable-diffusion-v1-5").to("cuda")

# Читаємо всі prompt'и з файлу prompts.txt (по одному на рядок)
with open("survival-test/prompts.txt", "r") as f:
    prompts = [line.strip() for line in f if line.strip()]

# Папка для збереження
save_dir = "/content/drive/MyDrive"
os.makedirs(save_dir, exist_ok=True)

# Генеруємо серію картинок
for i, prompt in enumerate(prompts, start=1):
    img = pipe(prompt).images[0]
    filename = os.path.join(save_dir, f"result_{i}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png")
    img.save(filename)
    display(img)
    print(f"✅ Збережено {filename}")
